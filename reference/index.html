API Reference

Module: runner

createRunner

Creates a task runner instance from a Lumio config object.

ParameterTypeDescription
configLumioConfigA validated Lumio configuration object.
optionsRunnerOptionsOptional runner settings.

Returns: Runner

import { createRunner } from "@lumio/runner";
import config from "./lumio.config";

const runner = createRunner(config, { verbose: true });
await runner.run("build");

defineConfig

Type-safe helper for authoring a Lumio configuration file.

ParameterTypeDescription
configLumioConfigConfiguration object to validate and return.

Returns: LumioConfig

Module: config

loadConfig

Reads and validates a lumio.config.ts file from the given directory.

ParameterTypeDescription
cwdstringDirectory to search for the config file. Defaults to process.cwd().

Returns: Promise<LumioConfig>

Types

LumioConfig

interface LumioConfig {
  tasks: Record<string, TaskDefinition>;
}

TaskDefinition

interface TaskDefinition {
  run: string;                // Shell command to execute
  dependsOn?: string[];       // Tasks that must complete first
  env?: Record<string, string>; // Environment variables
}

RunnerOptions

interface RunnerOptions {
  verbose?: boolean;   // Print each command before running it
  dryRun?: boolean;    // Resolve task order without executing
  cwd?: string;        // Working directory for all tasks
}

Runner

interface Runner {
  run(task: string): Promise<void>;
  runAll(options?: { parallel?: boolean }): Promise<void>;
  list(): TaskDefinition[];
}
